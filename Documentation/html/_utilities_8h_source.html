74  struct OnlineMeanEstimator
75  {
76  double n = 0.;
77  double mean = 0.;
78  double minValue = std::numeric_limits<double>::max (); // makes sure it will get
79  // overwritten by the
80  // first value
81  double maxValue = std::numeric_limits<double>::lowest (); // make sure it will
82  // get overwritten by
83  // the first value
84 
85  // From Knuth
86  // http://en.wikipedia.org/wiki/Algorithms_for_calculating_variance#Online_algorithm.
87  void update (double x)
88  {
89  mean = (mean * n + x) / (n + 1.0);
90  n += 1.0;
91 
92  minValue = std::min (minValue, x);
93  maxValue = std::max (maxValue, x);
94  }
95 
96  void update (const OnlineMeanEstimator& rhs)
97  {
98  if (rhs.n < 0.5) return;
99  mean = (mean * n + rhs.mean * rhs.n) / (n + rhs.n);
100  n += rhs.n;
101 
102  minValue = std::min (rhs.minValue, minValue);
103  maxValue = std::max (rhs.maxValue, maxValue);
104  }
105  };
